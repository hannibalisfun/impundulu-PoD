### Impundulu Submod — Scripted Triggers
### NOTE: is_impundulu_trigger already exists in PoD (POD_splat_triggers.txt) — do NOT redefine
### NOTE: has_impundulu_necromancy_trigger already exists in PoD (POD_necromancy_path_triggers.txt)

### Checks if the character has a courtier with the Bomkazi witch trait
has_bomkazi_courtier_trigger = {
	custom_tooltip = {
		text = IMP_has_bomkazi_courtier_tt
		any_courtier = {
			has_trait = bomkazi_witch
			is_alive = yes
		}
	}
}

### Checks if target character has any necromancy discipline
### Used for the Knowledge Exchange interaction to find valid targets
is_necromantic_vampire_trigger = {
	custom_tooltip = {
		text = IMP_is_necromantic_vampire_tt
		OR = {
			has_trait = necromancy
			has_trait = necromancyadvanced
		}
	}
}

### Checks if the character can craft Impundulu artifacts
### Used by artifact crafting decisions
can_craft_impundulu_artifact_trigger = {
	custom_tooltip = {
		text = IMP_can_craft_artifact_tt
		is_impundulu_trigger = yes
		is_vampire_trigger = yes
		OR = {
			has_impundulu_necromancy_trigger = yes
			has_any_necromancy_path = yes
		}
	}
}

### Checks if character has completed the major objective (either branch)
### Used by Legacy Tier 5 prerequisite
has_completed_withering_blood_trigger = {
	custom_tooltip = {
		text = IMP_has_completed_withering_blood_tt
		OR = {
			dynasty = { has_dynasty_modifier = covenant_of_life_and_death_modifier }
			has_character_modifier = ishologu_modifier
		}
	}
}

### Checks if character chose the Covenant Renewed branch
has_covenant_renewed_trigger = {
	dynasty = { has_dynasty_modifier = covenant_of_life_and_death_modifier }
}

### Checks if character chose the Ishologu Rite branch
has_ishologu_rite_trigger = {
	has_character_modifier = ishologu_modifier
}

### Checks if a character is a valid target for necromancy knowledge exchange
### Must be a vampire with necromancy, different bloodline, positive opinion
is_valid_knowledge_exchange_target_trigger = {
	custom_tooltip = {
		text = IMP_is_valid_exchange_target_tt
		is_vampire_trigger = yes
		is_necromantic_vampire_trigger = yes
		NOT = { is_impundulu_trigger = yes }
		opinion = {
			target = root
			value >= 20
		}
	}
}
