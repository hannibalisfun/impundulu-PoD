### Impundulu Submod — Scripted Effects

### Initialize Bomkazi cultivation for Impundulu players
### Called from on_action or introduction events
IMP_start_bomkazi_cultivation_effect = {
	if = {
		limit = {
			is_impundulu_trigger = yes
			is_playable_character = yes
			NOT = { has_character_flag = imp_bomkazi_cultivation_active }
		}
		trigger_event = {
			id = IMP_bomkazi.1
			days = 11
		}
	}
}

### Apply Covenant Renewed (objective branch A) rewards
### Called from the objective completion event
IMP_apply_covenant_renewed_effect = {
	custom_tooltip = IMP_covenant_renewed_reward_tt
	hidden_effect = {
		dynasty = {
			add_dynasty_modifier = {
				modifier = covenant_of_life_and_death_modifier
			}
		}
		add_prestige = major_prestige_value
		add_piety = major_piety_value
	}
}

### Apply Ishologu Rite (objective branch B) rewards
### Called from the objective completion event
IMP_apply_ishologu_rite_effect = {
	custom_tooltip = IMP_ishologu_rite_reward_tt
	hidden_effect = {
		add_character_modifier = {
			modifier = ishologu_modifier
		}
		add_prestige = massive_prestige_value
		dynasty = {
			add_dynasty_prestige = major_dynasty_prestige_value
		}
	}
}

### Mark major objective as completed in the global unavailable list
IMP_complete_withering_blood_objective_effect = {
	add_to_global_variable_list = {
		name = unavailable_unique_decisions
		target = flag:imp_withering_blood_objective_decision
	}
	POD_major_objective_finished_effect = yes
}
