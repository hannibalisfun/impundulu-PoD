### ==========================================
### MAJOR OBJECTIVE: "Cure the Withering Blood"
### ==========================================
### The Bomkazi mortal line is failing. Find a way to save it or transcend the dependency.
### 2-way branch: Covenant Renewed (preserve the bond) or Ishologu Rite (break free)

imp_withering_blood_objective_decision = {

	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_major_religion.dds"
	}
	decision_group_type = major

	ai_check_interval = 120

	desc = imp_withering_blood_objective_decision_desc

	is_shown = {
		is_playable_character = yes
		is_vampire_trigger = yes
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:imp_withering_blood_objective_decision
			}
		}
		is_impundulu_trigger = yes
	}

	is_valid = {
		### Must be House Head
		is_house_head = yes
		### Must have learned a necromancy path beyond Cenotaph/Haunting
		custom_tooltip = {
			text = imp_withering_blood_extra_necromancy_tt
			OR = {
				has_character_flag = has_giovanni_necromancy_flag
				has_character_flag = has_cappadocian_necromancy_flag
				has_character_flag = has_harbinger_necromancy_flag
				has_character_flag = has_lamia_necromancy_flag
				has_character_flag = has_nagaraja_necromancy_flag
				has_character_flag = has_rosselini_necromancy_flag
			}
		}
		### Realm size requirement
		realm_size >= 20
		### Must have a Bomkazi courtier
		has_bomkazi_courtier_trigger = yes
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_at_war = no
	}

	effect = {
		IMP_complete_withering_blood_objective_effect = yes

		show_as_tooltip = {
			custom_tooltip = imp_withering_blood_objective_decision_effect_tooltip
		}

		save_scope_as = victorious_impundulu
		trigger_event = IMP_objectives.1
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
