### Impundulu Submod — Knowledge Exchange Decision
### Allows Impundulu to exchange necromantic knowledge with foreign vampires
### Replaces the character interaction with a decision for better gameplay flow

imp_knowledge_exchange_decision = {

	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	}

	ai_check_interval = 0

	desc = imp_knowledge_exchange_decision_desc

	cooldown = { days = 18250 }

	is_shown = {
		is_impundulu_trigger = yes
		is_vampire_trigger = yes
		has_impundulu_necromancy_trigger = yes
	}

	is_valid = {
		### Must have a valid exchange target nearby
		custom_tooltip = {
			text = imp_requires_exchange_target_tt
			any_ruler = {
				is_vampire_trigger = yes
				is_necromantic_vampire_trigger = yes
				NOT = { is_impundulu_trigger = yes }
				opinion = {
					target = root
					value >= 20
				}
				is_alive = yes
			}
		}
		prestige >= 500
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_at_war = no
	}

	effect = {
		add_prestige = -500

		### Find and save a valid exchange target
		hidden_effect = {
			random_ruler = {
				limit = {
					is_vampire_trigger = yes
					is_necromantic_vampire_trigger = yes
					NOT = { is_impundulu_trigger = yes }
					opinion = {
						target = root
						value >= 20
					}
					is_alive = yes
				}
				save_scope_as = imp_exchange_target
			}
			trigger_event = { id = IMP_exchange.1 days = 360 }
		}
	}

	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = 0
	}
}
