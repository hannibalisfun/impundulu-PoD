namespace = IMP_objectives

### ==========================================
### MAJOR OBJECTIVE: "Cure the Withering Blood" — Completion Event
### ==========================================
### 2-way branch choice: Covenant Renewed or Ishologu Rite

IMP_objectives.1 = {
	type = character_event
	title = IMP_objectives.1.t
	desc = IMP_objectives.1.desc
	theme = default

	left_portrait = {
		character = scope:victorious_impundulu
		animation = personality_rational
	}

	trigger = {
		is_alive = yes
	}

	immediate = {
	}

	### Option A: "Covenant Renewed" — preserve the Bomkazi bond
	option = {
		name = IMP_objectives.1.a
		custom_tooltip = IMP_objectives.1.custom_a

		IMP_apply_covenant_renewed_effect = yes

		### Announce to other vampire players
		hidden_effect = {
			every_player = {
				limit = {
					is_vampire_trigger = yes
					NOT = { this = scope:victorious_impundulu }
				}
				trigger_event = { id = IMP_objectives.2 days = 3 }
			}
		}
	}

	### Option B: "The Ishologu Rite" — break free from the dependency
	option = {
		name = IMP_objectives.1.b
		custom_tooltip = IMP_objectives.1.custom_b

		IMP_apply_ishologu_rite_effect = yes

		### Announce to other vampire players
		hidden_effect = {
			every_player = {
				limit = {
					is_vampire_trigger = yes
					NOT = { this = scope:victorious_impundulu }
				}
				trigger_event = { id = IMP_objectives.2 days = 3 }
			}
		}
	}
}
