### Impundulu Submod — Artifact Crafting Events
### One event per artifact — the ritual itself
### Reference: POD artifact creation pattern

namespace = IMP_artifacts

### Event 1: Craft the Shew-Stone
IMP_artifacts.1 = {
	type = character_event
	title = IMP_artifacts.1.t
	desc = IMP_artifacts.1.desc
	theme = default

	left_portrait = {
		character = scope:imp_artifact_crafter
		animation = personality_rational
	}

	trigger = {
		is_alive = yes
	}

	immediate = {
		create_artifact = {
			name = artifact_shewstone_name
			description = artifact_shewstone_description
			modifier = artifact_shewstone_modifier
			type = miscellaneous
			visuals = small_box
			wealth = 50
			quality = 60
			save_scope_as = newly_created_artifact
			decaying = no
		}
		### Artifact created and scoped — no variable tracking needed
	}

	option = {
		name = IMP_artifacts.1.a
		custom_tooltip = IMP_shewstone_crafted_tt
	}
}

### Event 3: Craft the Witch Eye
IMP_artifacts.3 = {
	type = character_event
	title = IMP_artifacts.3.t
	desc = IMP_artifacts.3.desc
	theme = default

	left_portrait = {
		character = scope:imp_artifact_crafter
		animation = worry
	}

	trigger = {
		is_alive = yes
	}

	immediate = {
		create_artifact = {
			name = artifact_witch_eye_name
			description = artifact_witch_eye_description
			modifier = artifact_witch_eye_modifier
			type = miscellaneous
			visuals = small_box
			wealth = 90
			quality = 90
			save_scope_as = newly_created_artifact
			decaying = no
		}
		scope:newly_created_artifact = {
			set_variable = imp_witch_eye_artifact
		}
		### Apply the scarring modifier (self-mutilation cost)
		add_character_modifier = {
			modifier = imp_witch_eye_scarred_modifier
		}
	}

	option = {
		name = IMP_artifacts.3.a
		custom_tooltip = IMP_witch_eye_crafted_tt
	}
}
