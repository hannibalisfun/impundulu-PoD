### Impundulu Submod — Bomkazi Cultivation Events
### Reskin of Giovanni adoption mechanic (POD_giovanni_events.txt)
### Simplified: 6 events instead of Giovanni's 13

namespace = IMP_bomkazi

### Event 1: Cultivation Initiation
### Fires at game start for Impundulu players (via on_action or introduction)
### Player decides whether to seek a Bomkazi companion
IMP_bomkazi.1 = {
	type = character_event
	title = IMP_bomkazi.1.t
	desc = IMP_bomkazi.1.desc
	theme = default

	left_portrait = {
		character = root
		animation = personality_rational
	}

	trigger = {
		is_impundulu_trigger = yes
		is_playable_character = yes
		is_alive = yes
		NOT = { has_character_flag = imp_bomkazi_cultivation_active }
	}

	immediate = {
	}

	### Option A: Begin cultivation — seek a Bomkazi companion
	option = {
		name = IMP_bomkazi.1.a
		custom_tooltip = IMP_bomkazi_start_cultivation_tt
		hidden_effect = {
			add_character_flag = imp_bomkazi_cultivation_active
			trigger_event = { id = IMP_bomkazi.2 days = 3 }
		}
	}

	### Option B: Delay — not yet ready
	option = {
		name = IMP_bomkazi.1.b
		custom_tooltip = IMP_bomkazi_delay_cultivation_tt
		hidden_effect = {
			trigger_event = { id = IMP_bomkazi.1 days = { 7300 14600 } }
		}
	}

	### Option C: Refuse — I have no need for a witch
	option = {
		name = IMP_bomkazi.1.c
		custom_tooltip = IMP_bomkazi_refuse_cultivation_tt
		hidden_effect = {
			add_character_flag = imp_bomkazi_cultivation_refused
		}
	}
}

### Event 3: Hidden — Create the Bomkazi character based on outcome
IMP_bomkazi.3 = {
	hidden = yes
	type = character_event

	trigger = {
		is_alive = yes
	}

	immediate = {
		### Determine outcome: success, partial, or failure
		random_list = {
			70 = {
				### Full Success — create a proper Bomkazi witch
				create_character = {
					location = root.capital_province
					age = { 18 30 }
					culture = root.culture
					faith = root.faith
					gender_female_chance = 75
					trait = mortal
					learning = { 8 14 }
					diplomacy = { 4 8 }
					martial = { 2 6 }
					stewardship = { 4 8 }
					intrigue = { 4 10 }
					prowess = { 2 5 }
					random_traits = yes
					dynasty = dynasty:12000028
					save_scope_as = bomkazi_candidate
				}

				scope:bomkazi_candidate = {
					add_trait = bomkazi_witch
					add_trait = sorcerer
					add_trait = pure_blooded
					add_trait = revenant

					### Apply quality-specific bonuses
					if = {
						limit = { root = { has_character_flag = imp_bomkazi_seek_scholar } }
						add_trait = intellect_good_1
					}
					else_if = {
						limit = { root = { has_character_flag = imp_bomkazi_seek_healer } }
						add_trait = physique_good_1
					}
					else_if = {
						limit = { root = { has_character_flag = imp_bomkazi_seek_seer } }
						add_trait = beauty_good_1
					}
				}

				trigger_event = { id = IMP_bomkazi.4 days = 3 }
			}

			20 = {
				### Partial Success — create a Bomkazi apprentice
				create_character = {
					location = root.capital_province
					age = { 16 25 }
					culture = root.culture
					faith = root.faith
					gender_female_chance = 75
					trait = mortal
					learning = { 5 10 }
					diplomacy = { 3 7 }
					martial = { 2 5 }
					stewardship = { 3 7 }
					intrigue = { 3 8 }
					prowess = { 2 4 }
					random_traits = yes
					dynasty = dynasty:12000028
					save_scope_as = bomkazi_candidate
				}

				scope:bomkazi_candidate = {
					add_trait = bomkazi_apprentice
				}

				trigger_event = { id = IMP_bomkazi.5 days = 3 }
			}

			10 = {
				### Failure — no suitable candidate found
				trigger_event = { id = IMP_bomkazi.6 days = 3 }
			}
		}

		### Clean up flags
		remove_character_flag = imp_bomkazi_seek_scholar
		remove_character_flag = imp_bomkazi_seek_healer
		remove_character_flag = imp_bomkazi_seek_seer
	}

	option = {
		name = IMP_bomkazi.3.a
	}
}

### Event 4: Success — The Bomkazi witch arrives
IMP_bomkazi.4 = {
	type = character_event
	title = IMP_bomkazi.4.t
	desc = IMP_bomkazi.4.desc
	theme = default

	left_portrait = {
		character = root
		animation = schadenfreude
	}

	right_portrait = {
		character = scope:bomkazi_candidate
		animation = idle
	}

	trigger = {
		is_alive = yes
	}

	immediate = {
	}

	### Accept the Bomkazi witch into your court
	option = {
		name = IMP_bomkazi.4.a
		custom_tooltip = IMP_bomkazi_witch_arrives_tt

		scope:bomkazi_candidate = {
			set_relation_friend = root
		}
		if = {
			limit = {
				scope:bomkazi_candidate = {
					NOT = { is_courtier_of = root }
				}
			}
			add_courtier = scope:bomkazi_candidate
		}
		add_piety = minor_piety_value
		stress_impact = {
			base = medium_stress_impact_loss
		}

		### Allow the chain to restart in 20-40 years
		hidden_effect = {
			remove_character_flag = imp_bomkazi_cultivation_active
			trigger_event = { id = IMP_bomkazi.1 days = { 7300 14600 } }
		}
	}
}
