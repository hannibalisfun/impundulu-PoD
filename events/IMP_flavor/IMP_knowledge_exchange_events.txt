### Impundulu Submod — Knowledge Exchange Events
### 3-event chain: Approach → Negotiation → Exchange
### Reference: POD_disciplinetrain_events.txt

namespace = IMP_exchange

### Event 1: "The Approach" — the negotiation begins
IMP_exchange.1 = {
	type = character_event
	title = IMP_exchange.1.t
	desc = IMP_exchange.1.desc
	theme = default

	left_portrait = {
		character = root
		animation = personality_rational
	}

	right_portrait = {
		character = scope:imp_exchange_target
		animation = personality_rational
	}

	trigger = {
		is_alive = yes
	}

	immediate = {
	}

	### Option A: Offer gold — a straightforward exchange
	option = {
		name = IMP_exchange.1.a
		custom_tooltip = IMP_exchange_offer_gold_tt
		hidden_effect = {
			add_character_flag = imp_exchange_offered_gold
			trigger_event = { id = IMP_exchange.2 days = 30 }
		}
	}

	### Option B: Offer a favor — build the relationship
	option = {
		name = IMP_exchange.1.b
		custom_tooltip = IMP_exchange_offer_favor_tt
		hidden_effect = {
			add_character_flag = imp_exchange_offered_favor
			trigger_event = { id = IMP_exchange.2 days = 30 }
		}
	}

	### Option C: Share Impundulu secrets — mutual exchange
	option = {
		name = IMP_exchange.1.c
		custom_tooltip = IMP_exchange_offer_secrets_tt
		hidden_effect = {
			add_character_flag = imp_exchange_offered_secrets
			trigger_event = { id = IMP_exchange.2 days = 30 }
		}
	}
}

### Event 2: "The Negotiation" — outcome determined
IMP_exchange.2 = {
	type = character_event
	title = IMP_exchange.2.t
	desc = IMP_exchange.2.desc
	theme = default

	left_portrait = {
		character = root
		animation = personality_rational
	}

	right_portrait = {
		character = scope:imp_exchange_target
		animation = personality_rational
	}

	trigger = {
		is_alive = yes
	}

	immediate = {
		### Determine cost based on what was offered
		if = {
			limit = { has_character_flag = imp_exchange_offered_gold }
			remove_short_term_gold = 200
		}
		else_if = {
			limit = { has_character_flag = imp_exchange_offered_favor }
			### Create a hook for the target
			scope:imp_exchange_target = {
				add_hook = {
					target = root
					type = favor_hook
				}
			}
		}
		else_if = {
			limit = { has_character_flag = imp_exchange_offered_secrets }
			### The target also learns Impundulu necromancy
			scope:imp_exchange_target = {
				if = {
					limit = { has_impundulu_necromancy_trigger = no }
					add_character_flag = has_impundulu_necromancy_flag
				}
			}
		}

		### Clean up negotiation flags
		remove_character_flag = imp_exchange_offered_gold
		remove_character_flag = imp_exchange_offered_favor
		remove_character_flag = imp_exchange_offered_secrets
	}

	### Success — learn their path
	option = {
		name = IMP_exchange.2.a

		custom_tooltip = IMP_exchange_success_tt

		### Determine which path the target knows and learn it
		### Using show_as_tooltip so the player sees what they learned
		if = {
			limit = { scope:imp_exchange_target = { has_giovanni_necromancy_trigger = yes } }
			custom_tooltip = IMP_learned_giovanni_necromancy_tt
		}
		else_if = {
			limit = { scope:imp_exchange_target = { has_cappadocian_necromancy_trigger = yes } }
			custom_tooltip = IMP_learned_cappadocian_necromancy_tt
		}
		else_if = {
			limit = { scope:imp_exchange_target = { has_harbinger_necromancy_trigger = yes } }
			custom_tooltip = IMP_learned_harbinger_necromancy_tt
		}
		else_if = {
			limit = { scope:imp_exchange_target = { has_lamia_necromancy_trigger = yes } }
			custom_tooltip = IMP_learned_lamia_necromancy_tt
		}
		else_if = {
			limit = { scope:imp_exchange_target = { has_nagaraja_necromancy_trigger = yes } }
			custom_tooltip = IMP_learned_nagaraja_necromancy_tt
		}
		else_if = {
			limit = { scope:imp_exchange_target = { has_rosselini_necromancy_trigger = yes } }
			custom_tooltip = IMP_learned_rosselini_necromancy_tt
		}

		### Apply the actual effects
		hidden_effect = {
			if = {
				limit = { scope:imp_exchange_target = { has_giovanni_necromancy_trigger = yes } }
				if = {
					limit = { has_giovanni_necromancy_trigger = no }
					add_character_flag = has_giovanni_necromancy_flag
				}
			}
			else_if = {
				limit = { scope:imp_exchange_target = { has_cappadocian_necromancy_trigger = yes } }
				if = {
					limit = { has_cappadocian_necromancy_trigger = no }
					add_character_flag = has_cappadocian_necromancy_flag
				}
			}
			else_if = {
				limit = { scope:imp_exchange_target = { has_harbinger_necromancy_trigger = yes } }
				if = {
					limit = { has_harbinger_necromancy_trigger = no }
					add_character_flag = has_harbinger_necromancy_flag
				}
			}
			else_if = {
				limit = { scope:imp_exchange_target = { has_lamia_necromancy_trigger = yes } }
				if = {
					limit = { has_lamia_necromancy_trigger = no }
					add_character_flag = has_lamia_necromancy_flag
				}
			}
			else_if = {
				limit = { scope:imp_exchange_target = { has_nagaraja_necromancy_trigger = yes } }
				if = {
					limit = { has_nagaraja_necromancy_trigger = no }
					add_character_flag = has_nagaraja_necromancy_flag
				}
			}
			else_if = {
				limit = { scope:imp_exchange_target = { has_rosselini_necromancy_trigger = yes } }
				if = {
					limit = { has_rosselini_necromancy_trigger = no }
					add_character_flag = has_rosselini_necromancy_flag
				}
			}

			### Ensure character has necromancy trait
			if = {
				limit = {
					NOR = {
						has_trait = necromancy
						has_trait = necromancyadvanced
					}
				}
				add_trait = necromancy
			}

			### Add opinion modifier
			scope:imp_exchange_target = {
				add_opinion = {
					target = root
					modifier = fellow_scholar_of_death_modifier
				}
			}
		}

		add_prestige = minor_prestige_value
	}
}
